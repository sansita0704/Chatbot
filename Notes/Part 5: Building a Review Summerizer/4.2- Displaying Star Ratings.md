# Displaying Star Ratings

Create a new file `StarRating.tsx`:

```tsx
import { FaStar } from "react-icons/fa";
import { FaRegStar } from "react-icons/fa";

type Props = {
    value: number; // Rating from 0 to 5
};

const StarRating = ({ value }: Props) => {
    // Placeholder array for rendering 5 stars
    const placeholders = [1, 2, 3, 4, 5];

    return (
        <div className="flex gap-1 text-yellow-500">
            {placeholders.map((p) =>
                p <= value ? <FaStar key={p} /> : <FaRegStar key={p} />,
            )}
        </div>
    );
};

export default StarRating;
```

We map each placeholder to a star icon.

## Update `ReviewList`

Replace the text rating with the `StarRating` component.

```tsx
<div>
    <StarRating value={review.rating} />
</div>
```
